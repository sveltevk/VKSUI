Код

<script>
	import { FormField, FormItem, FormLayout, Group } from '@sveltevk/vksui';
</script>

<Group>
	<FormLayout
		on:submit={(e) => {
			e.preventDefault();
		}}
	>
		<FormItem>
			<FormField>
				<input type="text" class="CustomInput" placeholder="Кастомный инпут" />
			</FormField>
		</FormItem>
	</FormLayout>
</Group>
